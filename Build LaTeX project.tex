% ==================== SECTION 10 : RECOMMANDATIONS ====================
\section{Recommandations pour l'Execution}

\begin{enumerate}[leftmargin=*, label=\textbf{\arabic*.}]
    \item \textbf{Theorie des Contraintes :} Le stage est le \textcolor{riskred}{goulot}. Priorite absolue aux livrables terrain.
    
    \item \textbf{Principe de Pareto (80/20) :} 80\% de la valeur viendra de 20\% des activites (VSM, TRS, equilibrage).
    
    \item \textbf{Amelioration Continue (PDCA) :}
    \begin{itemize}
        \item \textbf{Plan :} Planning hebdomadaire (dimanche soir)
        \item \textbf{Do :} Execution quotidienne
        \item \textbf{Check :} Revue quotidienne (23h45)
        \item \textbf{Act :} Ajustements pour J+1
    \end{itemize}
    
    \item \textbf{SMED personnel :} Reduire les temps de transition entre activites (max 5 min).
    
    \item \textbf{Kanban visuel :} Utiliser un tableau Trello ou un tableau physique pour visualiser l'avancement.
    
    \item \textbf{Journal de bord :} Remplir chaque soir en 5 minutes. C'est votre memoire de stage en construction.
\end{enumerate}
